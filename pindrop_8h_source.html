44 /// @class Listener
45 ///
46 /// @brief An object whose distance from sounds determines their gain.
47 ///
48 /// The Listener class is a lightweight reference to a ListenerInternalState
49 /// which is managed by the AudioEngine. Multiple Listener objects may point to
50 /// the same underlying data.
51 class Listener {
52  public:
53  /// @brief Construct an uninitialized Listener.
54  ///
55  /// An uninitialized Listener can not have its location set or queried.
56  ///
57  /// To initialize the Listener, use <code>AudioEngine::AddListener();</code>
58  Listener() : state_(nullptr) {}
59 
60  explicit Listener(ListenerInternalState* state) : state_(state) {}
61 
62  /// @brief Uninitializes this Listener.
63  ///
64  /// Uninitializes this Listener. Note that this does not destroy the internal
65  /// state it references; it just removes this reference to it.
66  /// To destroy the Listener, use <code>AudioEngine::RemoveListener();</code>
67  void Clear();
68 
69  /// @brief Checks whether this Listener has been initialized.
70  ///
71  /// @return Returns true if this Listener is initialized.
72  bool Valid() const;
73 
74  /// @brief Get the location of this Listener.
75  ///
76  /// @return The location of this Listener.
77  mathfu::Vector<float, 3> Location() const;
78 
79  /// @brief Set the location of this Listener.
80  ///
81  /// @param location The new location of the Listener.
82  void SetLocation(const mathfu::Vector<float, 3>& location);
83 
84  ListenerInternalState* state() { return state_; }
85 
86  private:
87  ListenerInternalState* state_;
88 };

90 /// @class Channel
91 ///
92 /// @brief An object that represents a single channel of audio.
93 ///
94 /// The Channel class is a lightweight reference to a ChannelInternalState
95 /// which is managed by the AudioEngine. Multiple Channel objects may point to
96 /// the same underlying data.
97 class Channel {
98  public:
99  /// @brief Construct an uninitialized Listener.
100  ///
101  /// An uninitialized Construct can not have its location set or queried.
102  ///
103  /// To initialize the Channel, use <code>AudioEngine::PlaySound();</code>
104  Channel() : state_(nullptr) {}
105 
106  explicit Channel(ChannelInternalState* state) : state_(state) {}
107 
108  /// @brief Uninitializes this Channel.
109  ///
110  /// Uninitializes this Channel. Note that this does not stop the audio or
111  /// destroy the internal state it references; it just removes this reference
112  /// to it. To stop the Channel use <code>Channel::Stop();</code>
113  void Clear();
114 
115  /// @brief Checks whether this Channel has been initialized.
116  bool Valid() const;
117 
118  /// @brief Checks if the sound playing on a given Channel is playing.
119  ///
120  /// @return Whether the Channel is currently playing.
121  bool Playing() const;
122 
123  /// @brief Stop a channel.
124  ///
125  /// Stop this channel from playing. A sound will stop on its own if it not set
126  /// to loop. Looped audio must be explicitly stopped.
127  void Stop();
128 
129  /// @brief Get the location of this Channel.
130  ///
131  /// If the audio on this channel is not set to be Positional this property
132  /// does nothing.
133  ///
134  /// @return The location of this Channel.
135  const mathfu::Vector<float, 3> Location() const;
136 
137  /// @brief Set the location of this Channel.
138  ///
139  /// If the audio on this channel is not set to be Positional this property
140  /// does nothing.
141  ///
142  /// @param location The new location of the Channel.
143  void SetLocation(const mathfu::Vector<float, 3>& location);
144 
145  private:
146  ChannelInternalState* state_;
147 };
