46 /// @class Bus
47 ///
48 /// @brief An object representing one in a tree of buses. Buses are used to
49 /// adjust a set of channel gains in tandem.
50 ///
51 /// The Bus class is a lightweight reference to a BusInternalState which is
52 /// managed by the AudioEngine. There is always at least one bus, the master
53 /// bus, and any number of additional buses may be defined as well. Each bus can
54 /// be thought of as a node in a tree. The gain on a Bus is applied to all child
55 /// buses as well.
56 class Bus {
57  public:
58  /// @brief Construct an uninitialized Bus.
59  ///
60  /// An uninitialized Bus can not set or get any of it's fields.
61  ///
62  /// To initialize the Listener, use <code>AudioEngine::AddListener();</code>
63  Bus() : state_(nullptr) {}
64 
65  explicit Bus(BusInternalState* state) : state_(state) {}
66 
67  /// @brief Uninitializes this Bus.
68  ///
69  /// Uninitializes this Bus. Note that this does not destroy the internal
70  /// state it references; it just removes this reference to it.
71  void Clear();
72 
73  /// @brief Checks whether this Bus has been initialized.
74  ///
75  /// @return Returns true if this Bus is initialized.
76  bool Valid() const;
77 
78  /// @brief Sets the gain on this Bus.
79  ///
80  /// @param gain The new gain value.
81  void SetGain(float gain);
82 
83  /// @brief Returns the user specified gain on this bus.
84  ///
85  /// @return Returns the user specified gain.
86  float Gain() const;
87 
88  /// @brief Fades to <code>gain</code> over <code>duration</code> seconds.
89  //
90  /// @param gain The gain value to fade to.
91  /// @param duration The amount of time to take to reach the target gain.
92  void FadeTo(float gain, float duration);
93 
94  /// @brief Returns the final calculated gain on this bus.
95  ///
96  /// The FinalGain of a bus is the product of the gain specified in the Bus
97  /// definition file, the gain specified by the user, and the final gain of
98  /// this bus's parent bus.
99  ///
100  /// @return Returns the gain.
101  float FinalGain() const;
102 
103  BusInternalState* state() { return state_; }
104 
105  private:
106  BusInternalState* state_;
107 };

175 /// @class Channel
176 ///
177 /// @brief An object that represents a single channel of audio.
178 ///
179 /// The Channel class is a lightweight reference to a ChannelInternalState
180 /// which is managed by the AudioEngine. Multiple Channel objects may point to
181 /// the same underlying data.
182 class Channel {
183  public:
184  /// @brief Construct an uninitialized Listener.
185  ///
186  /// An uninitialized Construct can not have its location set or queried.
187  ///
188  /// To initialize the Channel, use <code>AudioEngine::PlaySound();</code>
189  Channel() : state_(nullptr) {}
190 
191  explicit Channel(ChannelInternalState* state) : state_(state) {}
192 
193  /// @brief Uninitializes this Channel.
194  ///
195  /// Uninitializes this Channel. Note that this does not stop the audio or
196  /// destroy the internal state it references; it just removes this reference
197  /// to it. To stop the Channel use <code>Channel::Stop();</code>
198  void Clear();
199 
200  /// @brief Checks whether this Channel has been initialized.
201  bool Valid() const;
202 
203  /// @brief Checks if the sound playing on a given Channel is playing.
204  ///
205  /// @return Whether the Channel is currently playing.
206  bool Playing() const;
207 
208  /// @brief Stop a channel.
209  ///
210  /// Stop this channel from playing. A sound will stop on its own if it not set
211  /// to loop. Looped audio must be explicitly stopped.
212  void Stop();
213 
214  /// @brief Pause a channel.
215  ///
216  /// Pause this channel. A paused channel may be resumed where it left off.
217  void Pause();
218 
219  /// @brief Resumes a paused channel.
220  ///
221  /// Resume this channel. If this channel was paused it will continue where it
222  /// left off.
223  void Resume();
224 
225  /// @brief Get the location of this Channel.
226  ///
227  /// If the audio on this channel is not set to be Positional this property
228  /// does nothing.
229  ///
230  /// @return The location of this Channel.
231  const mathfu::Vector<float, 3> Location() const;
232 
233  /// @brief Set the location of this Channel.
234  ///
235  /// If the audio on this channel is not set to be Positional this property
236  /// does nothing.
237  ///
238  /// @param location The new location of the Channel.
239  void SetLocation(const mathfu::Vector<float, 3>& location);
240 
241  /// @brief Sets the gain on this chanel.
242  ///
243  /// @param gain The new gain value.
244  void SetGain(float gain);
245 
246  /// @brief Returns the gain on this chanel.
247  ///
248  /// @return Returns the gain.
249  float Gain() const;
250 
251  private:
252  ChannelInternalState* state_;
253 };
